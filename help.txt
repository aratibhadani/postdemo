email: {
    type: DataTypes.STRING,
    allowNull: false,
    validate: {
      isUnique: (value, next) => {
        UserModel.findAll({
          where: { email: value },
          attributes: ['id'],
        })
          .then((user) => {
            if (user.length != 0)
              next(new Error('Email address already in use!'));
            next();
          })
          .catch((onError) => console.log(onError));
      },
    }
